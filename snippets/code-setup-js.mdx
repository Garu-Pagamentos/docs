const GARU_API_URL = 'https://garu.com.br/api';
const GARU_PAY_URL = 'https://garu.com.br/pay';

async function garuRequest(endpoint, options = {}) {
  const response = await fetch(`${GARU_API_URL}${endpoint}`, {
    ...options,
    headers: {
      'Authorization': `Bearer ${process.env.GARU_API_KEY}`,
      'Content-Type': 'application/json',
      ...options.headers
    }
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message || 'Erro na API');
  }

  if (response.status === 204) return null;
  return response.json();
}
